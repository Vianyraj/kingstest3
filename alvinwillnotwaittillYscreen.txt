<script>
// Global Modal Functions
function openModal(modalId) {
    const modal = document.getElementById(modalId);
    if (!modal) {
        console.error('Modal not found:', modalId);
        return;
    }
    
    modal.classList.remove('hidden');
    
    // Trigger animation after display
    setTimeout(() => {
        const modalContent = modal.querySelector('.bg-white');
        if (modalContent) {
            modalContent.classList.remove('scale-95', 'opacity-0');
            modalContent.classList.add('scale-100', 'opacity-100');
        }
    }, 10);
    
    // Prevent body scrolling
    document.body.style.overflow = 'hidden';
}

function closeModal(modalId) {
    const modal = document.getElementById(modalId);
    if (!modal) return;
    
    const modalContent = modal.querySelector('.bg-white');
    
    // Trigger close animation
    if (modalContent) {
        modalContent.classList.remove('scale-100', 'opacity-100');
        modalContent.classList.add('scale-95', 'opacity-0');
    }
    
    setTimeout(() => {
        modal.classList.add('hidden');
        document.body.style.overflow = 'auto';
    }, 300);
}

// Initialize everything when page loads
window.addEventListener('load', function() {
    // Set up modal close events
    document.querySelectorAll('[id$="-modal"]').forEach(modal => {
        modal.addEventListener('click', function(e) {
            if (e.target === this) {
                closeModal(this.id);
            }
        });
    });
    
    // Close modal with Escape key
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            document.querySelectorAll('[id$="-modal"]').forEach(modal => {
                if (!modal.classList.contains('hidden')) {
                    closeModal(modal.id);
                }
            });
        }
    });

    const observerOptions = {
        threshold: 0.15,
        rootMargin: '0px 0px -80px 0px'
    };

    const observer = new IntersectionObserver(function(entries) {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('visible');
            }
        });
    }, observerOptions);

    document.querySelectorAll('.scroll-reveal, .privilege-card').forEach(el => {
        observer.observe(el);
    });

    // New Perspective Carousel
    const carouselWrapper = document.getElementById('perspectiveCarouselWrapper');
    const slides = document.querySelectorAll('.perspective-carousel-slide');
    const prevBtn = document.getElementById('perspective-prev');
    const nextBtn = document.getElementById('perspective-next');

    let currentIndex = 0;
    const totalSlides = slides.length;

    function updateCarousel() {
        slides.forEach((slide, index) => {
            slide.classList.remove('active', 'prev-1', 'prev-2', 'next-1', 'next-2', 'hidden-slide');
            
            let newIndex = (index - currentIndex + totalSlides) % totalSlides;
            
            if (newIndex === 0) {
                slide.classList.add('active');
            } else if (newIndex === 1) {
                slide.classList.add('next-1');
            } else if (newIndex === 2) {
                slide.classList.add('next-2');
            } else if (newIndex === totalSlides - 1) {
                slide.classList.add('prev-1');
            } else if (newIndex === totalSlides - 2) {
                slide.classList.add('prev-2');
            } else {
                 slide.classList.add('hidden-slide');
            }
        });

        // Rotate the wrapper for a 3D effect
        const rotateAngle = -currentIndex * (360 / totalSlides);
        // carouselWrapper.style.transform = `rotateY(${rotateAngle}deg)`;
    }
    
    function showNext() {
        currentIndex = (currentIndex + 1) % totalSlides;
        updateCarousel();
    }

    function showPrev() {
        currentIndex = (currentIndex - 1 + totalSlides) % totalSlides;
        updateCarousel();
    }

    nextBtn.addEventListener('click', showNext);
    prevBtn.addEventListener('click', showPrev);

    slides.forEach(slide => {
        slide.addEventListener('click', () => {
            const slideIndex = parseInt(slide.dataset.index);
            if (slideIndex !== currentIndex) {
                currentIndex = slideIndex;
                updateCarousel();
            }
        });
    });

    // Autoplay
    let autoplayInterval = setInterval(showNext, 4000);

    carouselWrapper.addEventListener('mouseenter', () => clearInterval(autoplayInterval));
    carouselWrapper.addEventListener('mouseleave', () => autoplayInterval = setInterval(showNext, 4000));
    
    updateCarousel();

    // Testimonial Carousel
    const testimonialWrapper = document.getElementById('testimonial-carousel-wrapper');
    const testimonialSlides = document.querySelectorAll('#testimonial-carousel-wrapper .testimonial-card');
    const testimonialPrevBtn = document.getElementById('testimonial-prev');
    const testimonialNextBtn = document.getElementById('testimonial-next');
    const testimonialIndicatorsContainer = document.getElementById('testimonial-indicators');
    
    let testimonialCurrentIndex = 0;
    const testimonialTotalSlides = testimonialSlides.length;

    function createTestimonialIndicators() {
        for (let i = 0; i < testimonialTotalSlides; i++) {
            const button = document.createElement('button');
            button.classList.add('carousel-indicator');
            if (i === 0) button.classList.add('active');
            button.dataset.index = i;
            button.addEventListener('click', (e) => {
                testimonialCurrentIndex = parseInt(e.target.dataset.index);
                updateTestimonialCarousel();
            });
            testimonialIndicatorsContainer.appendChild(button);
        }
    }

    function updateTestimonialCarousel() {
        const offset = testimonialCurrentIndex * -100;
        
        testimonialSlides.forEach((slide, index) => {
            const relativeIndex = index - testimonialCurrentIndex;
            slide.style.transition = 'transform 0.6s ease, opacity 0.6s ease';
            
            if (relativeIndex === 0) { // Active slide
                slide.style.transform = `translate(-50%, -50%) translateX(0) translateZ(0) scale(1)`;
                slide.style.opacity = '1';
                slide.style.zIndex = '3';
            } else if (relativeIndex === 1 || relativeIndex === - (testimonialTotalSlides - 1)) { // Next slide
                slide.style.transform = `translate(-50%, -50%) translateX(70%) translateZ(-300px) scale(0.8)`;
                slide.style.opacity = '0.5';
                slide.style.zIndex = '2';
            } else if (relativeIndex === -1 || relativeIndex === (testimonialTotalSlides - 1)) { // Previous slide
                slide.style.transform = `translate(-50%, -50%) translateX(-70%) translateZ(-300px) scale(0.8)`;
                slide.style.opacity = '0.5';
                slide.style.zIndex = '2';
            } else { // Other slides
                 slide.style.transform = `translate(-50%, -50%) translateX(${relativeIndex > 0 ? '140%' : '-140%'}) translateZ(-600px) scale(0.6)`;
                slide.style.opacity = '0';
                slide.style.zIndex = '1';
            }
        });
        
        const indicators = testimonialIndicatorsContainer.querySelectorAll('.carousel-indicator');
        indicators.forEach((indicator, index) => {
            indicator.classList.toggle('active', index === testimonialCurrentIndex);
        });
    }
    
    testimonialNextBtn.addEventListener('click', () => {
        testimonialCurrentIndex = (testimonialCurrentIndex + 1) % testimonialTotalSlides;
        updateTestimonialCarousel();
    });
    
    testimonialPrevBtn.addEventListener('click', () => {
        testimonialCurrentIndex = (testimonialCurrentIndex - 1 + testimonialTotalSlides) % testimonialTotalSlides;
        updateTestimonialCarousel();
    });
    
    createTestimonialIndicators();
    updateTestimonialCarousel();

    let testimonialAutoplay = setInterval(() => {
        testimonialNextBtn.click();
    }, 5000);

    document.getElementById('testimonial-carousel-container').addEventListener('mouseenter', () => clearInterval(testimonialAutoplay));
    document.getElementById('testimonial-carousel-container').addEventListener('mouseleave', () => {
        testimonialAutoplay = setInterval(() => {
            testimonialNextBtn.click();
        }, 5000);
    });


    // Enhanced 360-degree panoramic image with auto-rotation
    const panoramaContainer = document.getElementById('panorama-container');
    const heroImage = document.getElementById('hero-360-image');
    let isDragging = false;
    let isHovering = false;
    let startX = 0;
    let currentX = 0;
    let velocityX = 0;
    let animationId = null;
    let autoRotateId = null;

    // Set initial position to center
    const imageWidth = heroImage.offsetWidth;
    const containerWidth = panoramaContainer.offsetWidth;
    currentX = -(imageWidth - containerWidth) / 2;
    heroImage.style.transform = `translateX(${currentX}px)`;

    // Auto-rotation function
    function autoRotate() {
        if (!isDragging && !isHovering) {
            currentX -= 0.5; // Slow rotation speed
            
            const minX = -(imageWidth - containerWidth);
            
            // Loop back when reaching the end
            if (currentX < minX) {
                currentX = 0;
            }
            
            heroImage.style.transform = `translateX(${currentX}px)`;
            autoRotateId = requestAnimationFrame(autoRotate);
        }
    }

    // Start auto-rotation
    autoRotate();

    panoramaContainer.addEventListener('mouseenter', function() {
        isHovering = true;
        if (autoRotateId) {
            cancelAnimationFrame(autoRotateId);
        }
    });

    panoramaContainer.addEventListener('mouseleave', function() {
        isHovering = false;
        if (!isDragging) {
            autoRotate();
        }
    });

    panoramaContainer.addEventListener('mousedown', function(e) {
        isDragging = true;
        startX = e.clientX - currentX;
        velocityX = 0;
        panoramaContainer.style.cursor = 'grabbing';
        if (animationId) {
            cancelAnimationFrame(animationId);
        }
        if (autoRotateId) {
            cancelAnimationFrame(autoRotateId);
        }
    });

    document.addEventListener('mousemove', function(e) {
        if (!isDragging) return;
        
        const x = e.clientX - startX;
        const maxX = 0;
        const minX = -(imageWidth - containerWidth);
        
        velocityX = x - currentX;
        currentX = Math.max(minX, Math.min(maxX, x));
        
        heroImage.style.transform = `translateX(${currentX}px)`;
    });

    document.addEventListener('mouseup', function() {
        if (isDragging) {
            isDragging = false;
            panoramaContainer.style.cursor = 'grab';
            
            // Add momentum/inertia effect
            function momentum() {
                if (Math.abs(velocityX) > 0.5) {
                    velocityX *= 0.95;
                    currentX += velocityX;
                    
                    const maxX = 0;
                    const minX = -(imageWidth - containerWidth);
                    currentX = Math.max(minX, Math.min(maxX, currentX));
                    
                    heroImage.style.transform = `translateX(${currentX}px)`;
                    animationId = requestAnimationFrame(momentum);
                } else {
                    if (!isHovering) {
                        autoRotate();
                    }
                }
            }
            momentum();
        }
    });

    // Touch events for mobile
    panoramaContainer.addEventListener('touchstart', function(e) {
        isDragging = true;
        isHovering = true;
        startX = e.touches[0].clientX - currentX;
        velocityX = 0;
        if (animationId) {
            cancelAnimationFrame(animationId);
        }
        if (autoRotateId) {
            cancelAnimationFrame(autoRotateId);
        }
    });

    document.addEventListener('touchmove', function(e) {
        if (!isDragging) return;
        
        const x = e.touches[0].clientX - startX;
        const maxX = 0;
        const minX = -(imageWidth - containerWidth);
        
        velocityX = x - currentX;
        currentX = Math.max(minX, Math.min(maxX, x));
        
        heroImage.style.transform = `translateX(${currentX}px)`;
    });

    document.addEventListener('touchend', function() {
        if (isDragging) {
            isDragging = false;
            isHovering = false;
            
            function momentum() {
                if (Math.abs(velocityX) > 0.5) {
                    velocityX *= 0.95;
                    currentX += velocityX;
                    
                    const maxX = 0;
                    const minX = -(imageWidth - containerWidth);
                    currentX = Math.max(minX, Math.min(maxX, currentX));
                    
                    heroImage.style.transform = `translateX(${currentX}px)`;
                    animationId = requestAnimationFrame(momentum);
                } else {
                    autoRotate();
                }
            }
            momentum();
        }
    });

    // Add parallax scrolling effect
    function parallaxScroll() {
        const parallaxSections = document.querySelectorAll('.parallax-section');
        
        parallaxSections.forEach(section => {
            const bgImage = section.querySelector('.parallax-bg-image');
            const overlay = section.querySelector('.parallax-overlay');
            const content = section.querySelector('.parallax-content');
            
            if (bgImage) {
                const scrolled = window.pageYOffset;
                const sectionTop = section.offsetTop;
                const sectionHeight = section.offsetHeight;
                const windowHeight = window.innerHeight;
                
                // Calculate if section is in viewport
                if (scrolled + windowHeight > sectionTop && scrolled < sectionTop + sectionHeight) {
                    const yPos = (scrolled - sectionTop) * 0.3;
                    bgImage.style.transform = `translateY(${yPos}px)`;
                    
                    if (overlay) {
                        overlay.style.transform = `translateY(${yPos * 0.5}px)`;
                    }
                    
                    if (content) {
                        content.style.transform = `translateY(${yPos * -0.1}px)`;
                    }
                }
            }
        });
    }
    
    window.addEventListener('scroll', parallaxScroll);
    parallaxScroll(); // Initial call

    // Other interactive images
    document.querySelectorAll('.interactive-image:not(#hero-360-image)').forEach(img => {
        img.addEventListener('mousemove', function(e) {
            const rect = this.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            const centerX = rect.width / 2;
            const centerY = rect.height / 2;
            
            const rotateX = (y - centerY) / centerY * 5;
            const rotateY = (x - centerX) / centerX * 5;
            
            this.style.transform = `rotateX(${-rotateX}deg) rotateY(${rotateY}deg) scale(1.02)`;
        });
        
        img.addEventListener('mouseleave', function() {
            this.style.transform = 'rotateX(0deg) rotateY(0deg) scale(1)';
        });
    });
});


// Interactive Mouse-Move Parallax for Feature Cards
function initMouseMoveParallax() {
    const featureCards = document.querySelectorAll('#feature-sports, #feature-fitness, #feature-entertainment, #feature-banquets, #feature-restaurant');
    
    featureCards.forEach(card => {
        const bgImage = card.querySelector('.parallax-bg-image');
        if (!bgImage) return;

        card.addEventListener('mousemove', (e) => {
            // Calculate mouse position relative to the card
            const rect = card.getBoundingClientRect();
            const x = e.clientX - rect.left; // 0 to width
            const y = e.clientY - rect.top;  // 0 to height
            
            // Calculate percentage (0 to 1)
            const xPercent = x / rect.width;
            const yPercent = y / rect.height;
            
            // Calculate movement range (e.g., Â±2%)
            const moveX = (xPercent - 0.5) * 4; // -2% to +2%
            const moveY = (yPercent - 0.5) * 4; // -2% to +2%
            
            // Apply the transform
            bgImage.style.transform = `translate(${moveX}%, ${moveY}%) scale(1.03)`;
        });
        
        card.addEventListener('mouseleave', () => {
            // Smoothly reset to the animated state when mouse leaves
            bgImage.style.transform = 'translate(0%, 0%) scale(1.02)';
        });
    });
}

// Call the function to initialize the effect
initMouseMoveParallax();
